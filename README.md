## –ú–µ—Ç–æ–¥ kNN

**–ê–ª–≥–æ—Ä–∏—Ç–º k –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π** (k nearest neighbors, kNN). –ú–µ—Ç—Ä–∏—á–µ—Å–∫–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º –¥–ª—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–π –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ –æ–±—ä–µ–∫—Ç–æ–≤.
–û–±—ä–µ–∫—Ç –ø—Ä–∏—Å–≤–∞–∏–≤–∞–µ—Ç—Å—è –∫ —Ç–æ–º—É –∫–ª–∞—Å—Å—É, –∫–æ—Ç–æ—Ä—ã–π —è–≤–ª—è–µ—Ç—Å—è –Ω–∞–∏–±–æ–ª–µ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω—ë–Ω–Ω—ã–º —Å—Ä–µ–¥–∏ k —Å–æ—Å–µ–¥–µ–π –¥–∞–Ω–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞, –∫–ª–∞—Å—Å—ã –∫–æ—Ç–æ—Ä—ã—Ö —É–∂–µ –∏–∑–≤–µ—Å—Ç–Ω—ã.


``` R
## –ü—Ä–∏–º–µ–Ω—è–µ–º –º–µ—Ç–æ–¥ kNN
kNN <- function(xl, z, k)
{
  ## –°–æ—Ä—Ç–∏—Ä—É–µ–º –≤—ã–±–æ—Ä–∫—É —Å–æ–≥–ª–∞—Å–Ω–æ –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º–æ–≥–æ –æ–±—ä–µ–∫—Ç–∞
	orderedXl <- sortObjectsByDist(xl, z)
	n <- dim(orderedXl)[2] - 1

  ## –ü–æ–ª—É—á–∞–µ–º –∫–ª–∞—Å—Å—ã –ø–µ—Ä–≤—ã—Ö k —Å–æ—Å–µ–¥–µ–π
	classes <- orderedXl[1:k, n + 1]

  ##–°–æ—Å—Ç–∞–≤–ª—è–µ–º —Ç–∞–±–ª–∏—Ü—É –≤—Å—Ç—Ä–µ—á–∞–µ–º–æ—Å—Ç–∏ –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞
	counts <- table(classes)

  ## –ù–∞—Ö–æ–¥–∏–º –∫–ª–∞—Å—Å, –∫–æ—Ç–æ—Ä—ã–π –¥–æ–º–∏–Ω–∏—Ä—É–µ—Ç —Å—Ä–µ–¥–∏ –ø–µ—Ä–≤—ã—Ö k —Å–æ—Å–µ–¥–µ–π
	class <- names(which.max(counts))
	return (class)
} 
```

``` R
Loo <- function(k,xl)
{
	sum <- 0
	for(i in 1:dim(xl)[1]){
		if(i==1){
				tmpXL <- xl[2:dim(xl)[1],]
			}
			else if (i==dim(xl)[1]) {
				tmpXL <- xl[1:dim(xl)[1]-1,]
			}
			else {
					
				tmpXL <- rbind(xl[1:i-1, ], xl[i+1:dim(xl)[1],])
			}

		xi <- c(xl[i,1], xl[i,2])
		class <-kNN(tmpXL,xi,k)
		if(class != xl[i,3])
		sum <- sum+1
	}
	return(sum)
}
```

## –ú–µ—Ç–æ–¥ kwNN

**–ú–µ—Ç–æ–¥ –≤–∑–≤–µ—à–µ–Ω–Ω—ã—Ö –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π.** –í –∑–∞–¥–∞—á–∞—Ö —Å —á–∏—Å–ª–æ–º –∫–ª–∞—Å—Å–æ–≤ 3 –∏ –±–æ–ª–µ–µ –Ω–µ—á—ë—Ç–Ω–æ—Å—Ç—å —É–∂–µ –Ω–µ –ø–æ–º–æ–≥–∞–µ—Ç, –∏ —Å–∏—Ç—É–∞—Ü–∏–∏ –Ω–µ–æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ—Å—Ç–∏ –≤—Å—ë —Ä–∞–≤–Ω–æ –º–æ–≥—É—Ç –≤–æ–∑–Ω–∏–∫–∞—Ç—å. –¢–æ–≥–¥–∞ i-–º—É —Å–æ—Å–µ–¥—É –ø—Ä–∏–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –≤–µ—Å w, –∫–∞–∫ –ø—Ä–∞–≤–∏–ª–æ, —É–±—ã–≤–∞—é—â–∏–π —Å —Ä–æ—Å—Ç–æ–º —Ä–∞–Ω–≥–∞ —Å–æ—Å–µ–¥–∞ i. –û–±—ä–µ–∫—Ç –æ—Ç–Ω–æ—Å–∏—Ç—Å—è –∫ —Ç–æ–º—É –∫–ª–∞—Å—Å—É, –∫–æ—Ç–æ—Ä—ã–π –Ω–∞–±–∏—Ä–∞–µ—Ç –±–æ–ª—å—à–∏–π —Å—É–º–º–∞—Ä–Ω—ã–π –≤–µ—Å —Å—Ä–µ–¥–∏ k –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π.

``` R
kwNN <- function(xl, z, k, q) 
{ 
  orderedXl <- sortObjectsByDist(xl, z) 
  n <- dim(orderedXl)[2] - 1 
  v1 <- c('setosa', 'versicolor', 'virginica')
  v2 <- c(0,0,0)
  
  for(i in 1:k){ 
    orderedXl[i, 4] = q^i 
  } 
  
  a=n+1 
  b=n+2 
  classes <- orderedXl[1:k, a:b]
  
  v2[1]=sum(classes[classes$Species=='setosa', 2])
  v2[2]=sum(classes[classes$Species=='versicolor', 2])
  v2[3]=sum(classes[classes$Species=='virginica', 2])
  
  amo <- cbind(v1,v2)
  
  class <- v1[which.max(v2)]
  return (class) 
}
```

``` R
      N=1/sqrt(2*pi)/sigm[i,j]*exp(-1/2*(x[j]-mu[i,j])^2/sigm[i,j]^2)
      scores[i] = scores[i] * N
    }
    amo[i,2]=scores[i]
  }
  class <- amo[,1][which.max(amo[,2])]
}
```

``` R
CovarianceMatrix = function(points, mu) {
    rows = dim(points)[1]
    cols = dim(points)[2]
    covar = matrix(0, cols, cols)
    for (i in 1:rows) {
        covar = covar + (t(points[i,] - mu) %*% (points[i,] - mu)) / (rows - 1)
    }
    return(covar)
}
```
